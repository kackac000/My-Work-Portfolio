
'' 데이터에서 선택한 기준열에 맞춰 여러개 파일로 분할 생성
Sub 파일분할()
    
    With Application
        .ScreenUpdating = False
        .EnableEvents = False
        .Application.CutCopyMode = False
        .Application.AskToUpdateLinks = False
    End With
    
    Dim startRow    '시작 행
    Dim startCol    '시작 열
    Dim curRow      '현재 로우
    Dim fileName    '저장할 파일 이름
    Dim dataCount   '분리할 데이터 갯수
    Dim endrow      '마지막 행
    Dim endcol      '마지막 열
    Dim count       '갯수
    Dim dir         '저장취치
    Dim activeSheetNumber   '시트번호
    
    On Error GoTo Err_StartCell '(시작위치)
    Set 비교대상 = Range(UserForm_FileSplit.lbCell)

    
    '(저장폴더 에러처리)
    Path = UserForm_FileSplit.lbLocation
    If (Len(Path) = 0 Or StrComp(Path, "파일을 저장할 위치를 지정해주세요.", vbTextCompare) = 0) Then
        GoTo Err_SaveFolder
    End If
        
    startCol = 비교대상.column
    startRow = 비교대상.Row
    curRow = startRow
        
    endrow = ActiveSheet.Cells(startRow, startCol).End(4).Row    '테이블의 특정열 마지막행
    endcol = ActiveSheet.Cells(startRow, startCol).End(2).column '테이블의 특정행 마지막열
    
    '' 시작 셀 마지막 셀 값들이 정상적인지 확인해본 창
    'MsgBox (startCol & ", " & startRow & ", " & endcol & ", " & endRow)
    
    'Range(Cells(startrow, startcol), Cells(endrow, endcol)).Sort key1:=Cells(4, 21), order1:=xlAscending

    
    fileName = Cells(startRow, startCol).Value
        
    For i = startRow To endrow   'i 현재 Row
        If Cells(i, startCol) = Cells(i + 1, startCol) Then
            dataCount = dataCount + 1
        Else
            count = count + 1
            '진행률 표시
            UserForm_FileSplit.lbValue = fileName
            UserForm_FileSplit.Repaint
            
            activeSheetNumber = ActiveSheet.index   '엑티브시트 번호
            
            '시트 복사
            If UserForm_FileSplit.cbAllSheet Then
                Sheets.Select
                Sheets.Copy
            Else
                ActiveSheet.Copy
            End If
            
            If FileExist(UserForm_FileSplit.lbLocation & "\" & fileName & ".xlsx") Then
                MsgBox "기준에 중복이 있어 파일쪼개기를 중단합니다. 정렬을 정확히 하고 진행하시기 바랍니다."
                Exit Sub
            End If
            
            On Error GoTo Err_FileName:
            Application.ActiveWorkbook.SaveAs fileName:=UserForm_FileSplit.lbLocation & "\" & fileName & ".xlsx"
                        
            If count = 1 Then
                Rows(i + 1 & ":" & endrow).Delete   '아래쪽 삭제
            Else
                If i <> endrow Then '아래쪽 삭제
                    Rows(i + 1 & ":" & endrow).Delete
                End If
                Rows(startRow & ":" & (i - dataCount)).Delete   '위쪽 삭제
            End If
    
            Application.ActiveWorkbook.Close True
            
            dataCount = 1
            fileName = Cells(i + 1, startCol).Value
        End If
    Next i
    
    '시트 선택
    Sheets(1).Select
    
    With Application
        .ScreenUpdating = True
        .EnableEvents = True
        .Application.CutCopyMode = True
        .Application.AskToUpdateLinks = True
    End With
    
    '완료 메시지창
    UserForm_FileSplit_Msg.StartUpPosition = 2
    UserForm_FileSplit_Msg.Top = 200
    UserForm_FileSplit_Msg.Left = 200
    UserForm_FileSplit_Msg.Show
    
    Exit Sub

Err_SaveFolder:
    MsgBox "저장위치를 지정해 주세요."
    Exit Sub
Err_StartCell:
    MsgBox "시작위치에 문제가 있습니다. 시작위치를 정확히 지정해주세요."
    Exit Sub
Err_FileName:
    MsgBox "저장중단(동일한 파일명이 존재하거나, 기준열에 파일명으로 사용할수없는 <>?[]:|* 등의 특수문자가 있거나, 파일명이 너무 길면 저장 안됨(경로포함 260자, 윈도우10기준))"
    Exit Sub
End Sub

Sub 시트저장()
    With Application
        .ScreenUpdating = False
        .EnableEvents = False
        .Application.CutCopyMode = False
        .Application.AskToUpdateLinks = False
    End With
        
    Dim filePath    '파일경로
    Dim fileName    '기존 파일명
    Dim name        '저장할 파일명
    Dim count       '갯수
    Dim toSheet As Worksheet
    Dim isCopy As Boolean   '복사대상여부
    Dim isHidden As Boolean '숨김 시트 여부
    count = Sheets.count
    filePath = ActiveWorkbook.Path
    fileName = ActiveWorkbook.name
    isCopy = True
    
    For i = 1 To count
        Set toSheet = ActiveWorkbook.Sheets(i)
        
        If toSheet.Visible = False Then
            isHidden = True
        Else
            isHidden = False
        End If
            
        If UserForm_SheetSave.cbHiddenFile Then '숨김파일도 포함이면
            toSheet.Visible = True
        Else    '숨김파일은 제거이면
            If toSheet.Visible Then
                isCopy = True
            Else
                isCopy = False
            End If
        End If
        
        If isCopy Then
            toSheet.Copy
            If UserForm_SheetSave.obFileSheet Then  '파일명 시트명
                name = fileName & " " & toSheet.name
            Else
                name = toSheet.name                 '시트명
            End If
            
            If UserForm_SheetSave.cbNumber Then     '숫자 포함 여부
                name = i & "_" & name
            End If
            
            Application.ActiveWorkbook.SaveAs filePath & "\" & name & ".xlsx"
            Application.ActiveWorkbook.Close True
        End If
        
        '숨김시트는 다시 숨김
        If isHidden Then
                toSheet.Visible = False
        End If
        
    Next i
    
    With Application
        .ScreenUpdating = True
        .EnableEvents = True
        .Application.CutCopyMode = True
        .Application.AskToUpdateLinks = True
    End With
    
    MsgBox "저장완료"
End Sub

'파일 존재여부 확인
Private Function FileExist(filePath As String) As Boolean

    Dim TestStr As String

    On Error Resume Next
      TestStr = dir(filePath)
    On Error GoTo 0
    
    If TestStr = "" Then
      FileExist = False
    Else
      FileExist = True
    End If

End Function

Sub 번호할당()
    
    Dim i, j, num, Line, endrow As Integer
    
    j = 1
    num = 1
    
    Line = Application.InputBox("파일내 제품 라인수", "라인수지정", Type:=1)
    endrow = ActiveSheet.Cells(1, 3).End(4).Row    '테이블의 특정열 마지막행
    
    For i = 1 To endrow - 3
    
        If j < Line + 1 Then
        
            ActiveSheet.Cells(i + 3, 24) = num
            j = j + 1
            
            If j = Line Then
            
                num = num + 1
                j = 1
                
            End If
            
            
        End If
        
    Next i
    
End Sub

Sub Labels_()

    Dim i As Long, r As Long, c As Long
    Dim rowCount As Long, dataRow As Long
    Dim labelsPerPage As Long: labelsPerPage = 16 '=============================== 2열 x 8행 3열 * 9행 (수원 16 부산 27)
    Dim outputSheet As Worksheet, dataSheet As Worksheet
    Dim pageOffset As Long
    Dim templateRange As Range
    Dim templateRange1 As Range

    Dim desktopPath As String
    desktopPath = CreateObject("WScript.Shell").SpecialFolders("Desktop")
    
    Dim TodayDate As String
    ' 오늘 날짜를 "YYYYMMDD" 형식으로 가져옴
    TodayDate = Format(Date, "yymmdd")
    
    Set dataSheet = Worksheets("데이터")
    Set outputSheet = Worksheets("라벨출력")
    Set templateRange = Worksheets("수원").Range("1:25") ' 서식 포함된 기본 템플릿 =========== 서식입력 (수원 1:25 부산 1:29)
    Set templateRange1 = Worksheets("수원").Range("A:D") ' 서식 포함된 기본 템플릿 =========== 서식입력 (수원 A:D 부산 A:F)
    
    rowCount = dataSheet.Cells(Rows.count, 5).End(xlUp).Row
    dataRow = 2
    pageOffset = 0

    outputSheet.Cells.ClearContents
    
    templateRange1.Copy
    outputSheet.Cells(pageOffset + 1, 1).PasteSpecial Paste:=xlPasteAllUsingSourceTheme

    Do While dataRow <= rowCount
        ' 1. 서식 복사
        templateRange.Copy
        outputSheet.Cells(pageOffset + 1, 1).PasteSpecial Paste:=xlPasteAllUsingSourceTheme
        'outputSheet.Cells(pageOffset + 1, 1).PasteSpecial Paste:=xlPasteValuesAndNumberFormats

        ' 2. 데이터 입력 (2열 × 8행 = 16개 기준, 수정 가능)
        For c = 0 To 1  ' 2열 0, 1 ============================================================ 입력 (2열; 1, 3열; 2)
            For r = 0 To 7  ' 8행 0 ~ 7 ======================================================= 입력 (8행; 7, 9행; 8)
                If dataRow <= rowCount Then
                    Dim baseRow As Long, baseCol As Long
                    baseRow = pageOffset + r * 3 + 1
                    baseCol = c * 2 + 1

                    With outputSheet
                        .Cells(baseRow, baseCol).Value = dataSheet.Cells(dataRow, 6).Value                      ' 제품코드
                        .Cells(baseRow, baseCol + 1).Formula = "*" & dataSheet.Cells(dataRow, 5).Value & "*"    ' 바코드"
                        .Cells(baseRow + 1, baseCol).Value = dataSheet.Cells(dataRow, 7).Value                  ' 제품명
                        .Cells(baseRow + 2, baseCol).Value = dataSheet.Cells(dataRow, 5).Value                  ' 로케이션
                    End With
                    dataRow = dataRow + 1
                End If
            Next r
        Next c

        ' 3. 다음 페이지 위치 (25행 간격으로 복사된 템플릿 반복)
        pageOffset = pageOffset + 25  '========================================================== 입력 (수원; 25, 부산; 29)
    Loop

    ' 인쇄 영역 지정
    outputSheet.PageSetup.PrintArea = "A1:D" & pageOffset   '==================================== 입력 (수원; A1:D, 부산; A1:F)

    ' PDF로 한 번에 저장
    outputSheet.ExportAsFixedFormat _
        Type:=xlTypePDF, _
        fileName:=desktopPath & "\" & TodayDate & " 라벨.pdf", _
        Quality:=xlQualityStandard, _
        IncludeDocProperties:=True, _
        IgnorePrintAreas:=False, _
        OpenAfterPublish:=True
        

    'MsgBox "PDF 생성 완료: 전체_서식라벨.pdf", vbInformation
    
    outputSheet.Columns("A:F").ClearContents
    

End Sub
